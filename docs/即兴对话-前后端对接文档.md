# 即兴对话——前后端对接文档（供 yiping）

## 功能描述

用户选两个角色 + 场景 → 两个角色自动你一句我一句聊起来，语音流式播放，直到停止。

---

## 后端需要新增

### POST /duet/start

**请求**：

```json
{
  "character_a": "huafei",
  "character_b": "zhenhuan",
  "context": "华妃和甄嬛在御花园偶遇",
  "starter": "huafei",
  "max_rounds": 10
}
```

**响应**：SSE 流，每个回合：角色文本 → 音频块 → 换对方继续

```
data: {"role":"huafei","text":"哟，这不是甄嬛妹妹吗？","emotion":"平静"}

data: {"audio":"base64...","dur":1.4}
data: {"audio":"base64...","dur":2.0}

data: {"role":"zhenhuan","text":"华妃娘娘安好。","emotion":"平静"}
data: {"audio":"base64...","dur":1.2}

...

event: done
```

### 实现参考

新建 `routers/duet.py`，在 `main.py` 中 `app.include_router(duet.router)`：

```python
# routers/duet.py
import json
from fastapi import APIRouter
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from services.llm_client import call_duet_generate   # 你需要新增
from services.tts_client import call_synthesize_stream  # 已有

router = APIRouter()

class DuetRequest(BaseModel):
    character_a: str
    character_b: str
    context: str = "两位角色在宫中相遇"
    starter: str
    max_rounds: int = 20


@router.post("/duet/start")
async def duet_start(req: DuetRequest):
    characters = [req.character_a, req.character_b]
    current = req.starter
    history = []

    async def event_stream():
        for _ in range(req.max_rounds):
            other = characters[1] if current == characters[0] else characters[0]

            # 1. LLM 生成下一句
            llm = await call_duet_generate({
                "my_character_id": current,
                "other_character_id": other,
                "context": req.context,
                "history": history,
                "my_turn": True,
            })
            text = llm["text"]
            emotion = llm.get("emotion", "平静")
            history.append({"role": current, "text": text})

            # 2. 推送文本给前端
            yield f"data: {json.dumps({'role': current, 'text': text, 'emotion': emotion}, ensure_ascii=False)}\n\n"

            # 3. TTS 流式合成 + 转发音频块
            async for chunk in call_synthesize_stream({
                "character_id": current,
                "text": text,
                "emotion": emotion,
                "engine": "cosyvoice",
            }):
                if "audio" in chunk:
                    yield f"data: {json.dumps({'audio': chunk['audio'], 'dur': chunk['dur']})}\n\n"

            # 换对方
            current = other

        yield "event: done\ndata: {}\n\n"

    return StreamingResponse(
        event_stream(),
        media_type="text/event-stream",
        headers={"Cache-Control": "no-cache, no-transform"},
    )
```

### 2. 即兴对话播放页

接收 SSE 流，逐回合显示 + 播放：

```javascript
async function startDuet(charA, charB, context, starter) {
  const abort = new AbortController()
  const audioCtx = new AudioContext()

  const resp = await fetch('/duet/start', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ character_a: charA, character_b: charB, context, starter }),
    signal: abort.signal,
  })

  const reader = resp.body.getReader()
  const decoder = new TextDecoder()
  let buffer = ''
  let nextTime = audioCtx.currentTime

  while (true) {
    const { done, value } = await reader.read()
    if (done) break
    buffer += decoder.decode(value, { stream: true })
    const lines = buffer.split('\n')
    buffer = lines.pop() || ''

    for (const line of lines) {
      if (!line.startsWith('data: ')) continue
      const data = JSON.parse(line.slice(6))

      if (data.role) {
        // 新增一条对话气泡，高亮说话角色
        addMessage(data.role, data.text, data.emotion)
      }
      if (data.audio) {
        // 流式播放（和现有 ChatPage 逻辑一样）
        const buf = await fetch('data:audio/wav;base64,' + data.audio).then(r => r.arrayBuffer())
        const decoded = await audioCtx.decodeAudioData(buf)
        const src = audioCtx.createBufferSource()
        src.buffer = decoded
        src.connect(audioCtx.destination)
        src.start(Math.max(audioCtx.currentTime, nextTime))
        nextTime += decoded.duration
      }
    }
  }
}
```
